58 static constexpr std::size_t Serialize(
59 const Message& msg, std::span<std::byte> output) noexcept {
60 std::size_t offset = StandardHeaderSize;
61 if (PayloadStartOffset > StandardHeaderSize) {
62 std::memset(output.data() + offset, 0,
63 PayloadStartOffset - StandardHeaderSize);
64 }
65 offset = PayloadStartOffset;
66
67 const MessageId msgId = Message::message_id;
68 const MessageId le_msgId = Crunch::LittleEndian(msgId);
69 std::memcpy(output.data() + offset, &le_msgId, sizeof(msgId));
70 offset += sizeof(msgId);
71
72 std::apply(
73 [&](const auto&... fields) {
74 ((offset = serialize_field(fields, output, offset)), ...);
75 },
76 msg.get_fields());
77 return offset;
78 }

80 template <typename Message>
87 [[nodiscard]] static constexpr auto Deserialize(
88 std::span<const std::byte> input, Message& msg) noexcept
89 -> std::optional<Error> {
90 std::size_t offset = PayloadStartOffset;
91
92 MessageId msg_id;
93 std::memcpy(&msg_id, input.data() + offset, sizeof(MessageId));
94 msg_id = Crunch::LittleEndian(msg_id);
95 if (msg_id != Message::message_id) {
96 return Error::invalid_message_id();
97 }
98 offset += sizeof(MessageId);
99
100 std::optional<Error> err = std::nullopt;
101 std::apply(
102 [&](auto&... fields) {
103 ((err.has_value()
104 ? void()
105 : [&] {
106 auto result =
107 deserialize_field(fields, input, offset);
108 if (result.has_value()) {
109 offset = result.value();
110 } else {
111 err = result.error();
112 }
113 }()),
114 ...);
115 },
116 msg.get_fields());
117 return err;
118 }

140 [[nodiscard]] static constexpr std::size_t calculate_padding(
141 std::size_t offset) noexcept {
142 const std::size_t align = std::min(sizeof(T), Alignment);
143 return (align - (offset % align)) % align;

242 template <typename T>
248 [[nodiscard]] static constexpr std::size_t calculate_array_end_offset(
249 std::size_t offset) noexcept {
250 using ValT = typename T::ValueType;
251 // Align for Length (uint32_t)
252 offset += calculate_padding<uint32_t>(offset) + sizeof(uint32_t);
253
254 // Calculate size of MaxSize elements
255 // Since element locations depend on alignment which depends on offset,
256 // we must iterate.
257 for (std::size_t i = 0; i < T::max_size; ++i) {
258 offset = calculate_value_end_offset<ValT>(offset);
259 }
260 return offset;

287 template <typename Field>
295 [[nodiscard]] static constexpr std::size_t serialize_field(
296 const Field& field, std::span<std::byte> output,
297 std::size_t offset) noexcept {
298 using ValueType = typename Field::FieldType;
299
300 // ArrayField doesn't have set byte - serialize directly
301 if constexpr (messages::is_array_field_v<Field>) {
302 return serialize_array(field, output, offset);
303 } else {
304 const bool set = field.set_;
305 output[offset++] = static_cast<std::byte>(set ? 1 : 0);
306
307 if (set) {
308 return serialize_value(field.value_, output, offset);
309 } else {
310 // Zero fill unset fields
311 const std::size_t end_offset =
312 calculate_value_end_offset<ValueType>(offset);
313 const std::size_t size = end_offset - offset;
314 std::memset(output.data() + offset, 0, size);
315 return end_offset;
316 }
317 }

328 [[nodiscard]] static constexpr std::size_t serialize_string(
329 const T& value, std::span<std::byte> output,
330 std::size_t offset) noexcept {
331 const std::size_t padding = calculate_padding<uint32_t>(offset);
332 if (padding > 0) {
333 std::memset(output.data() + offset, 0, padding);
334 offset += padding;
335 }
336
337 const uint32_t len = static_cast<uint32_t>(value.current_len_);
338 const uint32_t le_len = Crunch::LittleEndian(len);
339 std::memcpy(output.data() + offset, &le_len, sizeof(len));
340 offset += sizeof(len);
341
342 std::memcpy(output.data() + offset, value.buffer_.data(), T::max_size);
343 offset += T::max_size;
344 return offset;
345 }

402 template <typename T>
410 [[nodiscard]] static constexpr std::size_t serialize_array(
411 const T& value, std::span<std::byte> output,
412 std::size_t offset) noexcept {
413 using ValT = typename T::ValueType;
414 const std::size_t padding = calculate_padding<uint32_t>(offset);
415 if (padding > 0) {
416 std::memset(output.data() + offset, 0, padding);
417 offset += padding;
418 }
419
420 const uint32_t len = static_cast<uint32_t>(value.current_len_);
421 const uint32_t le_len = Crunch::LittleEndian(len);
422 std::memcpy(output.data() + offset, &le_len, sizeof(len));
423 offset += sizeof(len);
424
425 // Serialize elements
426 for (std::size_t i = 0; i < value.current_len_; ++i) {
427 offset = serialize_value(value.items_[i], output, offset);
428 }
429
430 // Zero fill remaining slots
431 for (std::size_t i = value.current_len_; i < T::max_size; ++i) {
432 const std::size_t end = calculate_value_end_offset<ValT>(offset);
433 std::memset(output.data() + offset, 0, end - offset);
434 offset = end;
435 }
436 return offset;
437 }

483 template <typename T>
492 [[nodiscard]] static constexpr auto deserialize_message(
493 T& value, bool set, std::span<const std::byte> input,
494 std::size_t offset) noexcept -> std::expected<std::size_t, Error> {
495 const std::size_t padding =
496 calculate_padding<std::byte[Alignment]>(offset);
497 offset += padding;
498
499 MessageId msg_id;
500 std::memcpy(&msg_id, input.data() + offset, sizeof(MessageId));
501 msg_id = Crunch::LittleEndian(msg_id);
502
503 if (set && msg_id != T::message_id) {
504 return std::unexpected(Error::invalid_message_id());
505 }
506 offset += sizeof(MessageId);
507
508 if (set) {
509 std::optional<Error> err = std::nullopt;
510 std::apply(
511 [&](auto&... sub_fields) {
512 ((err.has_value()
513 ? void()
514 : [&] {
515 auto result =
516 deserialize_field(sub_fields, input, offset);
517 if (result.has_value()) {
518 offset = result.value();
519 } else {
520 err = result.error();
521 }
522 }()),
523 ...);
524 },
525 value.get_fields());
526 if (err) {
527 return std::unexpected(err.value());
528 }
529 } else {
530 // Skip over submessage
531 offset += calculate_payload_size(T{});
532 }
533 return offset;
534 }

562 template <typename T>
571 [[nodiscard]] static constexpr auto deserialize_string(
572 T& value, bool set, std::span<const std::byte> input,
573 std::size_t offset) noexcept -> std::expected<std::size_t, Error> {
574 offset += calculate_padding<uint32_t>(offset);
575
576 uint32_t le_len;
577 std::memcpy(&le_len, input.data() + offset, sizeof(le_len));
578 offset += sizeof(le_len);
579 uint32_t len = Crunch::LittleEndian(le_len);
580
581 if (set) {
582 if (len > T::max_size) {
583 return std::unexpected(Error::capacity_exceeded(
584 0,
585 "deserialized string too long")); // No ID available here
586 }
587 std::memcpy(value.buffer_.data(), input.data() + offset,
588 T::max_size);
589 value.current_len_ = len;
590 } else {
591 value.clear();
592 }
593 offset += T::max_size;
594 return offset;
595 }

597 template <typename T>
606 [[nodiscard]] static constexpr auto deserialize_array(
607 T& value, bool set, std::span<const std::byte> input,
608 std::size_t offset) noexcept -> std::expected<std::size_t, Error> {
609 const auto array_end = calculate_array_end_offset<T>(offset);
610
611 if (!set) {
612 value.clear();
613 return array_end;
614 }
615
616 offset += calculate_padding<uint32_t>(offset);
617
618 uint32_t le_len;
619 std::memcpy(&le_len, input.data() + offset, sizeof(le_len));
620 offset += sizeof(le_len);
621 uint32_t len = Crunch::LittleEndian(le_len);
622
623 if (len > T::max_size) {
624 return std::unexpected(
625 Error::capacity_exceeded(0, "array capacity exceeded"));
626 }
627 value.current_len_ = len;
628
629 // Deserialize active elements
630 for (size_t i = 0; i < len; ++i) {
631 auto res = deserialize_value(value.items_[i], true, input, offset);
632 if (!res) {
633 return std::unexpected(res.error());
634 }
635 offset = res.value();
636 }
637
638 return array_end;
639 }
