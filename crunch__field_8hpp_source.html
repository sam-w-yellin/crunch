367 [[nodiscard]] constexpr auto Validate() const noexcept
368 -> std::optional<Error> {
369 // Validate each element
370 for (std::size_t i = 0; i < current_len_; ++i) {
371 std::optional<Error> err;
372 if constexpr (HasValidateWithId<ElementType>) {
373 err = items_[i].Validate(Id);
374 } else if constexpr (HasValidateNoId<ElementType>) {
375 err = items_[i].Validate();
376 }
377 if (err.has_value()) {
378 return err;
379 }
380 }
381
382 // Run array-level validators
383 if constexpr (sizeof...(Validators) > 0) {
384 for (const auto& result : {Validators::Check(*this, Id)...}) {
385 if (result.has_value()) {
386 return result;
387 }
388 }
389 }
390
391 return std::nullopt;
392 }

396 if (current_len_ != other.current_len_) {
397 return false;
398 }
399 return std::equal(items_.begin(), items_.begin() + current_len_,
400 other.items_.begin());
401 }

405 auto end() const noexcept { return items_.begin() + current_len_; }
406 auto begin() noexcept { return items_.begin(); }
407 auto end() noexcept { return items_.begin() + current_len_; }
