387 [[nodiscard]] constexpr auto Validate() const noexcept
388 -> std::optional<Error> {
389 // Validate each element
390 for (std::size_t i = 0; i < current_len_; ++i) {
391 std::optional<Error> err;
392 if constexpr (HasValidateWithId<ElementType>) {
393 err = items_[i].Validate(Id);
394 } else if constexpr (HasValidateNoId<ElementType>) {
395 err = items_[i].Validate();
396 }
397 if (err.has_value()) {
398 return err;
399 }
400 }

402 // Run array-level validators
403 if constexpr (sizeof...(Validators) > 0) {
404 for (const auto& result : {Validators::Check(*this, Id)...}) {
405 if (result.has_value()) {
406 return result;
407 }
408 }
409 }

411 return std::nullopt;
412 }

420 other.items_.begin());
421 }

467class MapField {
468 static_assert(ValidElementType<KeyField>, "Invalid KeyField type");
469 static_assert(ValidElementType<ValueField>, "Invalid ValueField type");
470 // static_assert(sizeof...(Validators) >= 0); // Always true
471 public:
472 static constexpr FieldId field_id = Id;
473 // cppcheck-suppress unusedStructMember
474 static constexpr std::size_t max_size = MaxSize;
475
476 using KeyType = field_value_type_t<KeyField>;
477 using ValueType = field_value_type_t<ValueField>;
478 using PairType = std::pair<KeyField, ValueField>;
479 using FieldType = MapField;
480
481 constexpr MapField() noexcept = default;
482
490 constexpr std::optional<Error> insert(const KeyType& key,
491 const ValueType& value) noexcept {
492 std::optional<Error> err;
493
494 // Validate Key
495 if constexpr (Crunch::fields::is_scalar_v<KeyField> ||
496 Crunch::fields::is_string_v<KeyField>) {
497 // For scalar/string, KeyType is value type. Validate against
498 // validator.
499 err = KeyField::Validate(key, 0);
500 } else {
501 // For complex, KeyType is field type. Validate against internal
502 // rules.
503 err = key.Validate();
504 }
505 if (err) {
506 return err;
507 }
508
509 // Validate Value
510 if constexpr (Crunch::fields::is_scalar_v<ValueField> ||
511 Crunch::fields::is_string_v<ValueField>) {
512 err = ValueField::Validate(value, 0);
513 } else {
514 err = value.Validate();
515 }
516 if (err) {
517 return err;
518 }
519
520 if (current_len_ >= MaxSize) {
521 return Error::capacity_exceeded(Id, "map capacity exceeded");
522 }
523
524 // Check for duplicate key
525 if (at(key).has_value()) {
526 return Error::validation(Id, "Duplicate key in map");
527 }
528
529 auto& pair = items_[current_len_];
530
531 // 3. Store Key
532 if constexpr (Crunch::fields::is_scalar_v<KeyField>) {
533 pair.first.set_without_validation(key);
534 } else if constexpr (Crunch::fields::is_string_v<KeyField>) {
535 static_cast<void>(pair.first.set(key));
536 } else {
537 pair.first = key;
538 }
539
540 // 4. Store Value
541 if constexpr (Crunch::fields::is_scalar_v<ValueField>) {
542 pair.second.set_without_validation(value);
543 } else if constexpr (Crunch::fields::is_string_v<ValueField>) {
544 static_cast<void>(pair.second.set(value));
545 } else {
546 pair.second = value;
547 }
548
549 current_len_++;
550 return std::nullopt;
551 }
552
553 // Overload for inserting std::pair
554 constexpr std::optional<Error> insert(
555 const std::pair<KeyType, ValueType>& p) noexcept {
556 return insert(p.first, p.second);
557 }

564 // cppcheck-suppress unusedFunction
565 constexpr bool remove(const KeyType& key) noexcept {
566 for (std::size_t i = 0; i < current_len_; ++i) {
567 const auto& stored_key_field = items_[i].first;
568 if (key_equals(stored_key_field, key)) {
569 // Found key at index i.
570 // Clear element at i
571 items_[i].first.clear();
572 items_[i].second.clear();
573
574 // Shift remaining elements down
575 for (std::size_t j = i; j < current_len_ - 1; ++j) {
576 items_[j] = items_[j + 1];
577 }
578 // Clear last element (now duplicated at penultimate pos)
579 // Actually the move assignment above likely did a copy.
580 // We should clear the old last element slot to be safe/clean.
581 items_[current_len_ - 1].first.clear();
582 items_[current_len_ - 1].second.clear();
583
584 current_len_--;
585 return true;
586 }
587 }
588 return false;
589 }

596 constexpr std::optional<ValueField*> at(const KeyType& key) noexcept {
597 for (std::size_t i = 0; i < current_len_; ++i) {
598 // We need to extract the underlying value from the KeyField to
599 // compare
600 const auto& stored_key_field = items_[i].first;
601 // Assuming scalar/string keys have .get() returning optional or
602 // value
603 if (key_equals(stored_key_field, key)) {
604 return &items_[i].second;
605 }
606 }
607 return std::nullopt;
608 }

636 if (const auto err = items_[i].first.Validate(); err) {

659 [[nodiscard]] constexpr bool operator==(

664
665 // I don't have a faster way to do this given

672
673 if (!other.has_entry(my_key, my_val)) {
674 return false;
675 }

677 return true;
678 }

682 auto begin() noexcept { return items_.begin(); }

689 constexpr bool has_entry(const KeyField& key, const ValueField& val) const {
690 auto it = std::find_if(

692 return it != end() && it->second == val;

698 Crunch::fields::is_string_v<KeyField>) {
