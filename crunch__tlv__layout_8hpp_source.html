93 template <typename Message>
94 [[nodiscard]] static constexpr std::size_t Serialize(
95 const Message& msg, std::span<std::byte> output) noexcept {
96 std::size_t offset = Crunch::StandardHeaderSize;
97
98 const std::size_t length_field_offset = offset;
99 offset += sizeof(uint32_t);
100
101 const std::size_t payload_start = offset;
102 offset = serialize_fields_helper(msg.get_fields(), output, offset);
103
104 const std::size_t payload_size = offset - payload_start;
105 const uint32_t le_len =
106 Crunch::LittleEndian(static_cast<uint32_t>(payload_size));
107 std::memcpy(output.data() + length_field_offset, &le_len,
108 sizeof(uint32_t));
109
110 return offset;
111 }

120 template <typename Message>
121 [[nodiscard]] static constexpr auto Deserialize(
122 std::span<const std::byte> input, Message& msg) noexcept
123 -> std::optional<Error> {
124 std::size_t offset = Crunch::StandardHeaderSize;
125
126 if (offset + sizeof(uint32_t) > input.size()) {
127 return Error::deserialization("buffer too small for tlv length");
128 }
129
130 uint32_t le_len;
131 std::memcpy(&le_len, input.data() + offset, sizeof(uint32_t));
132 uint32_t payload_len = Crunch::LittleEndian(le_len);
133 offset += sizeof(uint32_t);
134
135 if (offset + payload_len > input.size()) {
136 return Error::deserialization("tlv length exceeds buffer");
137 }
138
139 return deserialize_message_payload(
140 input.subspan(0, offset + payload_len), msg, offset);
141 }

152 [[nodiscard]] static constexpr std::size_t write_tag(
153 FieldId id, WireType wt, std::span<std::byte> output,
154 std::size_t offset) noexcept {
155 const uint32_t tag = (static_cast<uint32_t>(id) << WireTypeBits) |
156 static_cast<uint8_t>(wt);
157 return offset + Varint::encode(tag, output, offset);
158 }

252 template <typename T>
253 [[nodiscard]] static consteval std::size_t
254 calculate_max_value_size() noexcept {
255 if constexpr (Crunch::fields::is_scalar_v<T>) {
256 // Just the varint value, no tag
257 return Varint::max_size;
258 } else if constexpr (Crunch::fields::is_string_v<T>) {
259 // [Length][Data]
260 return Varint::max_size + T::max_size;
261 } else if constexpr (Crunch::messages::HasCrunchMessageInterface<T>) {
262 // [Length][NestedFields]
263 return Varint::max_size + calculate_max_message_size<T>();
264 } else if constexpr (Crunch::messages::is_array_field_v<T>) {
265 return calculate_max_array_field_size<typename T::ValueType>(
266 T::max_size);
267 } else if constexpr (Crunch::messages::is_map_field_v<T>) {
268 using KeyType = typename T::PairType::first_type;
269 using ValueType = typename T::PairType::second_type;
270 return calculate_max_map_field_size<KeyType, ValueType>(
271 T::max_size);
272 }
273 std::unreachable();
274 }

337 template <typename T>
338 [[nodiscard]] static constexpr std::size_t serialize_scalar_value(
339 const T& value, std::span<std::byte> output,
340 std::size_t offset) noexcept {
341 uint64_t encoded_val = 0;
342 if constexpr (std::is_same_v<T, bool>) {
343 encoded_val = value ? 1 : 0;
344 } else if constexpr (std::is_floating_point_v<T>) {
345 if constexpr (sizeof(T) == 4) {
346 encoded_val =
347 static_cast<uint64_t>(std::bit_cast<uint32_t>(value));
348 } else {
349 encoded_val = std::bit_cast<uint64_t>(value);
350 }
351 } else {
352 encoded_val = static_cast<uint64_t>(
353 std::bit_cast<std::make_unsigned_t<T>>(value));
354 }
355 return offset + Varint::encode(encoded_val, output, offset);
356 }

367 [[nodiscard]] static constexpr std::size_t serialize_string_value(
368 const T& value, std::span<std::byte> output,
369 std::size_t offset) noexcept {
370 auto sv = value.get();
371 offset += Varint::encode(sv.size(), output, offset);
372 std::memcpy(output.data() + offset, sv.data(), sv.size());
373 return offset + sv.size();

390 [[nodiscard]] static constexpr std::size_t fixup_length_prefix(
391 std::span<std::byte> output, std::size_t len_offset,
392 std::size_t content_start, std::size_t offset) noexcept {
393 const std::size_t content_size = offset - content_start;
394 const std::size_t actual_len_varint_size = Varint::size(content_size);
395 if (actual_len_varint_size < Varint::max_size) {
396 const std::size_t shift = Varint::max_size - actual_len_varint_size;
397 std::memmove(output.data() + len_offset + actual_len_varint_size,
398 output.data() + content_start, content_size);
399 offset -= shift;
400 }
401 Varint::encode(content_size, output, len_offset);
402 return offset;
403 }

436 template <typename FieldT>
437 [[nodiscard]] static constexpr std::size_t serialize_array_content(
438 const FieldT& field, std::span<std::byte> output,
439 std::size_t offset) noexcept {
440 using ElemT = typename FieldT::ValueType;
441
442 // Write element count
443 offset += Varint::encode(field.size(), output, offset);
444
445 // Serialize elements
446 for (const auto& item : field) {
447 if constexpr (Crunch::fields::is_scalar_v<ElemT>) {
448 offset = serialize_scalar_value(item.get(), output, offset);
449 } else if constexpr (Crunch::fields::is_string_v<ElemT>) {
450 offset = serialize_string_value(item, output, offset);
451 } else if constexpr (Crunch::messages::HasCrunchMessageInterface<
452 ElemT>) {
453 offset = serialize_nested_message(item, output, offset);
454 } else if constexpr (Crunch::messages::is_array_field_v<ElemT>) {
455 offset = serialize_array_value(item, output, offset);
456 } else if constexpr (Crunch::messages::is_map_field_v<ElemT>) {
457 offset = serialize_map_value(item, output, offset);
458 }
459 }
460 return offset;
461 }

543 template <typename FieldT>
544 [[nodiscard]] static constexpr std::size_t serialize_map_content(
545 const FieldT& field, std::span<std::byte> output,
546 std::size_t offset) noexcept {
547 using KeyFieldT = typename FieldT::PairType::first_type;
548 using ValueFieldT = typename FieldT::PairType::second_type;
549
550 // Write entry count
551 offset += Varint::encode(field.size(), output, offset);
552
553 // Serialize key-value pairs without tags
554 return std::accumulate(
555 field.begin(), field.end(), offset,
556 [&](std::size_t off, const auto& item) {
557 off = serialize_value_without_tag<KeyFieldT>(item.first, output,
558 off);
559 return serialize_value_without_tag<ValueFieldT>(item.second,
560 output, off);
561 });
562 }

617 template <typename FieldT>
618 [[nodiscard]] static constexpr std::size_t serialize_field(
619 const FieldT& field, std::span<std::byte> output,
620 std::size_t offset) noexcept {
621 bool is_set = false;
622 if constexpr (Crunch::messages::is_array_field_v<FieldT> ||
623 Crunch::messages::is_map_field_v<FieldT>) {
624 is_set = !field.empty();
625 } else {
626 is_set = field.set_;
627 }
628
629 if (!is_set) {
630 return offset;
631 }
632
633 const FieldId id = field.field_id;
634
635 if constexpr (Crunch::messages::is_array_field_v<FieldT>) {
636 return serialize_array_field(field, output, offset);
637 } else if constexpr (Crunch::messages::is_map_field_v<FieldT>) {
638 return serialize_map_field(field, output, offset);
639 } else {
640 using ValueType = typename detail::ext<FieldT>::type;
641 if constexpr (Crunch::fields::is_scalar_v<ValueType>) {
642 offset = write_tag(id, WireType::Varint, output, offset);
643 return serialize_scalar_value(field.value_.get(), output,
644 offset);
645 } else if constexpr (Crunch::fields::is_string_v<ValueType>) {
646 offset =
647 write_tag(id, WireType::LengthDelimited, output, offset);
648 return serialize_string_value(field.value_, output, offset);
649 } else if constexpr (Crunch::messages::HasCrunchMessageInterface<
650 ValueType>) {
651 offset =
652 write_tag(id, WireType::LengthDelimited, output, offset);
653 return serialize_nested_message(field.value_, output, offset);
654 }
655 }
656 return offset;
657 }

687 template <typename ElemT>
688 [[nodiscard]] static constexpr std::optional<Error>
689 deserialize_scalar_array(typename ElemT::ValueType& val_out,
690 std::span<const std::byte> input,
691 std::size_t& offset) noexcept {
692 using T = typename ElemT::ValueType;
693 const auto res = Varint::decode(input, offset);
694 if (!res) {
695 return Error::deserialization("invalid varint in packed");
696 }
697 offset += res->second;
698
699 if constexpr (std::is_same_v<T, bool>) {
700 val_out = (res->first != 0);
701 } else if constexpr (std::is_floating_point_v<T>) {
702 if constexpr (sizeof(T) == 4) {
703 val_out = std::bit_cast<T>(static_cast<uint32_t>(res->first));
704 } else {
705 val_out = std::bit_cast<T>(res->first);
706 }
707 } else {
708 val_out = std::bit_cast<T>(
709 static_cast<std::make_unsigned_t<T>>(res->first));
710 }
711 return std::nullopt;
712 }

721 [[nodiscard]] static constexpr std::expected<std::size_t, Error>
722 read_length_prefix(std::span<const std::byte> input, std::size_t& offset,
723 const char* error_msg) noexcept {
724 const auto len_res = Varint::decode(input, offset);
725 if (!len_res) {
726 return std::unexpected(Error::deserialization(error_msg));
727 }
728 offset += len_res->second;
729 std::size_t len = static_cast<std::size_t>(len_res->first);
730 if (offset + len > input.size()) {
731 return std::unexpected(Error::deserialization("buffer underflow"));
732 }
733 return len;
734 }

766 [[nodiscard]] static constexpr std::optional<Error>
767 deserialize_string_value(ElemT& val, std::span<const std::byte> input,
768 std::size_t& offset) noexcept {
769 auto len_result =
770 read_length_prefix(input, offset, "invalid string length");
771 if (!len_result) {
772 return len_result.error();
773 }
774 std::size_t len = *len_result;
775 std::string_view sv(
776 reinterpret_cast<const char*>(input.data() + offset), len);
777 offset += len;
778 return val.set(sv);

901 template <typename FieldT>
902 [[nodiscard]] static constexpr std::optional<Error>
903 deserialize_array_elements(FieldT& field, std::span<const std::byte> input,
904 std::size_t& offset,
905 std::size_t end_offset) noexcept {
906 using ElemT = typename FieldT::ValueType;
907
908 // Read count
909 const auto count_res = Varint::decode(input, offset);
910 if (!count_res) {
911 return Error::deserialization("invalid array count");
912 }
913 offset += count_res->second;
914 const std::size_t count = static_cast<std::size_t>(count_res->first);
915
916 // Deserialize elements
917 for (std::size_t i = 0; i < count; ++i) {
918 ElemT elem{};
919 if (const auto err =
920 deserialize_value_without_tag<ElemT>(elem, input, offset)) {
921 return err;
922 }
923 if constexpr (Crunch::fields::is_scalar_v<ElemT>) {
924 if (const auto err = field.add(elem.get())) {
925 return err;
926 }
927 } else {
928 if (const auto err = field.add(elem)) {
929 return err;
930 }
931 }
932 }
933 return std::nullopt;
934 }

945 [[nodiscard]] static constexpr std::optional<Error>
946 deserialize_message_payload(std::span<const std::byte> input, Message& msg,
947 std::size_t offset) noexcept {
948 while (offset < input.size()) {
949 const auto tag_res = Varint::decode(input, offset);
950 if (!tag_res) {
951 return Error::deserialization("invalid tag varint");
952 }
953 const auto [tag, tag_bytes] = *tag_res;
954
955 offset += tag_bytes;
956
957 const uint32_t field_id =
958 static_cast<uint32_t>(tag >> WireTypeBits);
959 const WireType wire_type = static_cast<WireType>(tag & 0x07);
960
961 bool found = false;
962 std::optional<Error> err = visit_fields_tlv(
963 msg.get_fields(), static_cast<FieldId>(field_id), wire_type,
964 input, offset, found);
965
966 if (err) {
967 return err;
968 }
969 if (!found) {
970 err.emplace(Error::deserialization("unknown fields present"));
971 return err;
972 }
973 }
974 return std::nullopt;
975 }

989 template <typename FieldT>
990 [[nodiscard]] static constexpr std::optional<Error> deserialize_array_field(
991 FieldT& field, WireType wire_type, std::span<const std::byte> input,
992 std::size_t& offset) noexcept {
993 if (wire_type != WireType::LengthDelimited) {
994 return Error::deserialization("array must be length delimited");
995 }
996
997 // Read total length
998 const auto len_res = Varint::decode(input, offset);
999 if (!len_res) {
1000 return Error::deserialization("invalid array length");
1001 }
1002 offset += len_res->second;
1003 const std::size_t len = static_cast<std::size_t>(len_res->first);
1004 if (offset + len > input.size()) {
1005 return Error::deserialization("array underflow");
1006 }
1007
1008 auto subspan = input.subspan(offset, len);
1009 std::size_t sub_offset = 0;
1010 if (const auto err =
1011 deserialize_array_elements(field, subspan, sub_offset, len)) {
1012 return err;
1013 }
1014
1015 offset += len;
1016 return std::nullopt;

1028 template <typename FieldT>
1029 [[nodiscard]] static constexpr std::optional<Error>
1030 deserialize_scalar_field(FieldT& field, WireType wire_type,
1031 std::span<const std::byte> input,
1032 std::size_t& offset) noexcept {
1033 using ValueType = typename detail::ext<FieldT>::type;
1034 using T = typename ValueType::ValueType;
1035
1036 if (wire_type != WireType::Varint) {
1037 return Error::deserialization("scalar must be varint");
1038 }
1039
1040 const auto res = Varint::decode(input, offset);
1041 if (!res) {
1042 return Error::deserialization("invalid varint");
1043 }
1044 offset += res->second;
1045
1046 T val;
1047 if constexpr (std::is_same_v<T, bool>) {
1048 val = (res->first != 0);
1049 } else if constexpr (std::is_floating_point_v<T>) {
1050 if constexpr (sizeof(T) == 4) {
1051 val = std::bit_cast<T>(static_cast<uint32_t>(res->first));
1052 } else {
1053 val = std::bit_cast<T>(res->first);
1054 }
1055 } else {
1056 val = std::bit_cast<T>(
1057 static_cast<std::make_unsigned_t<T>>(res->first));
1058 }
1059 // Deserialize will validate fields after the entire message is
1060 // deserialized
1061 field.set_without_validation(val);
1062 return std::nullopt;
1063 }

1119 }
1120
1121 const auto len_res = Varint::decode(input, offset);

1123 return Error::deserialization("invalid length");
1124 }

1177 [[nodiscard]] static constexpr std::optional<Error>
1178 deserialize_map_elements(FieldT& field, std::span<const std::byte> input,
1179 std::size_t& offset,
1180 std::size_t end_offset) noexcept {
1181 using KeyFieldT = typename FieldT::PairType::first_type;
1182 using ValueFieldT = typename FieldT::PairType::second_type;
1183
1184 // Read count
1185 const auto count_res = Varint::decode(input, offset);
1186 if (!count_res) {
1187 return Error::deserialization("invalid map count");
1188 }
1189 offset += count_res->second;
1190 const std::size_t count = static_cast<std::size_t>(count_res->first);
1191
1192 // Deserialize key-value pairs
1193 for (std::size_t i = 0; i < count; ++i) {
1194 KeyFieldT key_field{};
1195 ValueFieldT val_field{};
1196
1197 if (const auto err = deserialize_value_without_tag<KeyFieldT>(
1198 key_field, input, offset)) {
1199 return err;
1200 }
1201 if (const auto err = deserialize_value_without_tag<ValueFieldT>(
1202 val_field, input, offset)) {
1203 return err;
1204 }
1205
1206 if (const auto err = field.insert(extract_map_value(key_field),
1207 extract_map_value(val_field))) {
1208 return err;
1209 }
1210 }
1211 return std::nullopt;

1265 template <typename FieldT>
1266 [[nodiscard]] static constexpr std::optional<Error> deserialize_field_value(
1267 FieldT& field, WireType wire_type, std::span<const std::byte> input,
1268 std::size_t& offset) noexcept {

1270 return deserialize_array_field(field, wire_type, input, offset);
1271 } else if constexpr (Crunch::messages::is_map_field_v<FieldT>) {
1272 return deserialize_map_field(field, wire_type, input, offset);
1273 } else {

1285 }
1286 }
1287 return std::nullopt;

1303 template <typename Tuple, std::size_t I = 0>
1304 [[nodiscard]] static constexpr std::optional<Error> visit_fields_tlv(
1305 Tuple&& fields_tuple, FieldId target_id, WireType wt,
1306 std::span<const std::byte> input, std::size_t& offset,
1307 bool& found) noexcept {
1308 using TupleT = std::remove_cvref_t<Tuple>;
1309 if constexpr (I < std::tuple_size_v<TupleT>) {
1310 auto& f = std::get<I>(fields_tuple);
1311 if (f.field_id == target_id) {
1312 found = true;
1313 return deserialize_field_value(f, wt, input, offset);
1314 }
1315 return visit_fields_tlv<Tuple, I + 1>(
1316 std::forward<Tuple>(fields_tuple), target_id, wt, input, offset,
1317 found);
1318 }
1319 return std::nullopt;
1320 }
