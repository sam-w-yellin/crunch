93 template <typename Message>
94 [[nodiscard]] static constexpr std::size_t Serialize(
95 const Message& msg, std::span<std::byte> output) noexcept {
96 std::size_t offset = Crunch::StandardHeaderSize;
97
98 // Write Message ID
99 const MessageId msgId = Message::message_id;
100 const MessageId le_msgId = Crunch::LittleEndian(msgId);
101 std::memcpy(output.data() + offset, &le_msgId, sizeof(msgId));
102 offset += sizeof(msgId);
103
104 const std::size_t length_field_offset = offset;
105 offset += sizeof(uint32_t);
106
107 const std::size_t payload_start = offset;
108 offset = serialize_fields_helper(msg.get_fields(), output, offset);
109
110 const std::size_t payload_size = offset - payload_start;
111 const uint32_t le_len =
112 Crunch::LittleEndian(static_cast<uint32_t>(payload_size));
113 std::memcpy(output.data() + length_field_offset, &le_len,
114 sizeof(uint32_t));
115
116 return offset;
117 }

126 template <typename Message>
127 [[nodiscard]] static constexpr auto Deserialize(
128 std::span<const std::byte> input, Message& msg) noexcept
129 -> std::optional<Error> {
130 std::size_t offset = Crunch::StandardHeaderSize;
131
132 // Validate Message ID
133 if (offset + sizeof(MessageId) > input.size()) {
134 return Error::deserialization("buffer too small for message id");
135 }
136 MessageId msg_id;
137 std::memcpy(&msg_id, input.data() + offset, sizeof(MessageId));
138 msg_id = Crunch::LittleEndian(msg_id);
139 if (msg_id != Message::message_id) {
140 return Error::invalid_message_id();
141 }
142 offset += sizeof(MessageId);
143
144 if (offset + sizeof(uint32_t) > input.size()) {
145 return Error::deserialization("buffer too small for tlv length");
146 }
147
148 uint32_t le_len;
149 std::memcpy(&le_len, input.data() + offset, sizeof(uint32_t));
150 uint32_t payload_len = Crunch::LittleEndian(le_len);
151 offset += sizeof(uint32_t);
152
153 if (offset + payload_len > input.size()) {
154 return Error::deserialization("tlv length exceeds buffer");
155 }
156
157 return deserialize_message_payload(
158 input.subspan(0, offset + payload_len), msg, offset);
159 }

170 [[nodiscard]] static constexpr std::size_t write_tag(
171 FieldId id, WireType wt, std::span<std::byte> output,
172 std::size_t offset) noexcept {
173 const uint32_t tag = (static_cast<uint32_t>(id) << WireTypeBits) |
174 static_cast<uint8_t>(wt);
175 return offset + Varint::encode(tag, output, offset);
176 }

270 template <typename T>
271 [[nodiscard]] static consteval std::size_t
272 calculate_max_value_size() noexcept {
273 if constexpr (Crunch::fields::is_scalar_v<T>) {
274 // Just the varint value, no tag
275 return Varint::max_size;
276 } else if constexpr (Crunch::fields::is_string_v<T>) {
277 // [Length][Data]
278 return Varint::max_size + T::max_size;
279 } else if constexpr (Crunch::messages::HasCrunchMessageInterface<T>) {
280 // [Length][NestedFields]
281 return Varint::max_size + calculate_max_message_size<T>();
282 } else if constexpr (Crunch::messages::is_array_field_v<T>) {
283 return calculate_max_array_field_size<typename T::ValueType>(
284 T::max_size);
285 } else if constexpr (Crunch::messages::is_map_field_v<T>) {
286 using KeyType = typename T::PairType::first_type;
287 using ValueType = typename T::PairType::second_type;
288 return calculate_max_map_field_size<KeyType, ValueType>(
289 T::max_size);
290 }
291 std::unreachable();
292 }

355 template <typename T>
356 [[nodiscard]] static constexpr std::size_t serialize_scalar_value(
357 const T& value, std::span<std::byte> output,
358 std::size_t offset) noexcept {
359 uint64_t encoded_val = 0;
360 if constexpr (std::is_same_v<T, bool>) {
361 encoded_val = value ? 1 : 0;
362 } else if constexpr (std::is_floating_point_v<T>) {
363 if constexpr (sizeof(T) == 4) {
364 encoded_val =
365 static_cast<uint64_t>(std::bit_cast<uint32_t>(value));
366 } else {
367 encoded_val = std::bit_cast<uint64_t>(value);
368 }
369 } else {
370 encoded_val = static_cast<uint64_t>(
371 std::bit_cast<std::make_unsigned_t<T>>(value));
372 }
373 return offset + Varint::encode(encoded_val, output, offset);
374 }

385 [[nodiscard]] static constexpr std::size_t serialize_string_value(
386 const T& value, std::span<std::byte> output,
387 std::size_t offset) noexcept {
388 auto sv = value.get();

390 std::memcpy(output.data() + offset, sv.data(), sv.size());
391 return offset + sv.size();

408 [[nodiscard]] static constexpr std::size_t fixup_length_prefix(
409 std::span<std::byte> output, std::size_t len_offset,
410 std::size_t content_start, std::size_t offset) noexcept {
411 const std::size_t content_size = offset - content_start;
412 const std::size_t actual_len_varint_size = Varint::size(content_size);
413 if (actual_len_varint_size < Varint::max_size) {
414 const std::size_t shift = Varint::max_size - actual_len_varint_size;
415 std::memmove(output.data() + len_offset + actual_len_varint_size,
416 output.data() + content_start, content_size);
417 offset -= shift;
418 }
419 Varint::encode(content_size, output, len_offset);
420 return offset;
421 }

454 template <typename FieldT>
455 [[nodiscard]] static constexpr std::size_t serialize_array_content(
456 const FieldT& field, std::span<std::byte> output,
457 std::size_t offset) noexcept {
458 using ElemT = typename FieldT::ValueType;
459
460 // Write element count
461 offset += Varint::encode(field.size(), output, offset);
462
463 // Serialize elements
464 for (const auto& item : field) {
465 if constexpr (Crunch::fields::is_scalar_v<ElemT>) {
466 offset = serialize_scalar_value(item.get(), output, offset);
467 } else if constexpr (Crunch::fields::is_string_v<ElemT>) {
468 offset = serialize_string_value(item, output, offset);
469 } else if constexpr (Crunch::messages::HasCrunchMessageInterface<
470 ElemT>) {
471 offset = serialize_nested_message(item, output, offset);
472 } else if constexpr (Crunch::messages::is_array_field_v<ElemT>) {
473 offset = serialize_array_value(item, output, offset);
474 } else if constexpr (Crunch::messages::is_map_field_v<ElemT>) {
475 offset = serialize_map_value(item, output, offset);
476 }
477 }
478 return offset;
479 }

561 template <typename FieldT>
562 [[nodiscard]] static constexpr std::size_t serialize_map_content(
563 const FieldT& field, std::span<std::byte> output,
564 std::size_t offset) noexcept {
565 using KeyFieldT = typename FieldT::PairType::first_type;
566 using ValueFieldT = typename FieldT::PairType::second_type;
567
568 // Write entry count
569 offset += Varint::encode(field.size(), output, offset);
570
571 // Serialize key-value pairs without tags
572 return std::accumulate(
573 field.begin(), field.end(), offset,
574 [&](std::size_t off, const auto& item) {
575 off = serialize_value_without_tag<KeyFieldT>(item.first, output,
576 off);
577 return serialize_value_without_tag<ValueFieldT>(item.second,
578 output, off);
579 });

635 template <typename FieldT>
636 [[nodiscard]] static constexpr std::size_t serialize_field(
637 const FieldT& field, std::span<std::byte> output,
638 std::size_t offset) noexcept {
639 bool is_set = false;
640 if constexpr (Crunch::messages::is_array_field_v<FieldT> ||
641 Crunch::messages::is_map_field_v<FieldT>) {
642 is_set = !field.empty();
643 } else {
644 is_set = field.set_;
645 }
646
647 if (!is_set) {
648 return offset;
649 }
650
651 const FieldId id = field.field_id;
652
653 if constexpr (Crunch::messages::is_array_field_v<FieldT>) {
654 return serialize_array_field(field, output, offset);
655 } else if constexpr (Crunch::messages::is_map_field_v<FieldT>) {
656 return serialize_map_field(field, output, offset);
657 } else {
658 using ValueType = typename detail::ext<FieldT>::type;
659 if constexpr (Crunch::fields::is_scalar_v<ValueType>) {
660 offset = write_tag(id, WireType::Varint, output, offset);
661 return serialize_scalar_value(field.value_.get(), output,
662 offset);
663 } else if constexpr (Crunch::fields::is_string_v<ValueType>) {
664 offset =
665 write_tag(id, WireType::LengthDelimited, output, offset);
666 return serialize_string_value(field.value_, output, offset);
667 } else if constexpr (Crunch::messages::HasCrunchMessageInterface<
668 ValueType>) {
669 offset =
670 write_tag(id, WireType::LengthDelimited, output, offset);
671 return serialize_nested_message(field.value_, output, offset);
672 }
673 }
674 return offset;
675 }

705 template <typename ElemT>
706 [[nodiscard]] static constexpr std::optional<Error>
707 deserialize_scalar_array(typename ElemT::ValueType& val_out,
708 std::span<const std::byte> input,
709 std::size_t& offset) noexcept {
710 using T = typename ElemT::ValueType;
711 const auto res = Varint::decode(input, offset);
712 if (!res) {
713 return Error::deserialization("invalid varint in packed");
714 }
715 offset += res->second;
716
717 if constexpr (std::is_same_v<T, bool>) {
718 val_out = (res->first != 0);
719 } else if constexpr (std::is_floating_point_v<T>) {
720 if constexpr (sizeof(T) == 4) {
721 val_out = std::bit_cast<T>(static_cast<uint32_t>(res->first));
722 } else {
723 val_out = std::bit_cast<T>(res->first);
724 }
725 } else {
726 val_out = std::bit_cast<T>(
727 static_cast<std::make_unsigned_t<T>>(res->first));
728 }
729 return std::nullopt;
730 }

739 [[nodiscard]] static constexpr std::expected<std::size_t, Error>
740 read_length_prefix(std::span<const std::byte> input, std::size_t& offset,
741 const char* error_msg) noexcept {
742 const auto len_res = Varint::decode(input, offset);
743 if (!len_res) {
744 return std::unexpected(Error::deserialization(error_msg));
745 }
746 offset += len_res->second;
747 std::size_t len = static_cast<std::size_t>(len_res->first);
748 if (offset + len > input.size()) {
749 return std::unexpected(Error::deserialization("buffer underflow"));
750 }
751 return len;
752 }

919 template <typename FieldT>
920 [[nodiscard]] static constexpr std::optional<Error>
921 deserialize_array_elements(FieldT& field, std::span<const std::byte> input,
922 std::size_t& offset,
923 std::size_t end_offset) noexcept {
924 using ElemT = typename FieldT::ValueType;
925
926 // Read count
927 const auto count_res = Varint::decode(input, offset);
928 if (!count_res) {
929 return Error::deserialization("invalid array count");
930 }
931 offset += count_res->second;
932 const std::size_t count = static_cast<std::size_t>(count_res->first);
933
934 // Deserialize elements
935 for (std::size_t i = 0; i < count; ++i) {
936 ElemT elem{};
937 if (const auto err =
938 deserialize_value_without_tag<ElemT>(elem, input, offset)) {
939 return err;
940 }
941 if constexpr (Crunch::fields::is_scalar_v<ElemT>) {
942 if (const auto err = field.add(elem.get())) {
943 return err;
944 }
945 } else {
946 if (const auto err = field.add(elem)) {
947 return err;
948 }
949 }
950 }
951 return std::nullopt;
952 }

962 template <typename Message>
963 [[nodiscard]] static constexpr std::optional<Error>
964 deserialize_message_payload(std::span<const std::byte> input, Message& msg,
965 std::size_t offset) noexcept {
966 while (offset < input.size()) {
967 const auto tag_res = Varint::decode(input, offset);
968 if (!tag_res) {
969 return Error::deserialization("invalid tag varint");
970 }
971 const auto [tag, tag_bytes] = *tag_res;
972
973 offset += tag_bytes;
974
975 const uint32_t field_id =
976 static_cast<uint32_t>(tag >> WireTypeBits);
977 const WireType wire_type = static_cast<WireType>(tag & 0x07);
978
979 bool found = false;
980 std::optional<Error> err = visit_fields_tlv(
981 msg.get_fields(), static_cast<FieldId>(field_id), wire_type,
982 input, offset, found);
983
984 if (err) {
985 return err;
986 }
987 if (!found) {
988 err.emplace(Error::deserialization("unknown fields present"));
989 return err;
990 }
991 }
992 return std::nullopt;
993 }

1007 template <typename FieldT>
1008 [[nodiscard]] static constexpr std::optional<Error> deserialize_array_field(
1009 FieldT& field, WireType wire_type, std::span<const std::byte> input,
1010 std::size_t& offset) noexcept {
1011 if (wire_type != WireType::LengthDelimited) {
1012 return Error::deserialization("array must be length delimited");
1013 }
1014
1015 // Read total length
1016 const auto len_res = Varint::decode(input, offset);
1017 if (!len_res) {
1018 return Error::deserialization("invalid array length");
1019 }
1020 offset += len_res->second;
1021 const std::size_t len = static_cast<std::size_t>(len_res->first);
1022 if (offset + len > input.size()) {
1023 return Error::deserialization("array underflow");
1024 }
1025
1026 auto subspan = input.subspan(offset, len);
1027 std::size_t sub_offset = 0;
1028 if (const auto err =
1029 deserialize_array_elements(field, subspan, sub_offset, len)) {
1030 return err;
1031 }
1032
1033 offset += len;
1034 return std::nullopt;

1046 template <typename FieldT>
1047 [[nodiscard]] static constexpr std::optional<Error>
1048 deserialize_scalar_field(FieldT& field, WireType wire_type,
1049 std::span<const std::byte> input,
1050 std::size_t& offset) noexcept {
1051 using ValueType = typename detail::ext<FieldT>::type;
1052 using T = typename ValueType::ValueType;
1053
1054 if (wire_type != WireType::Varint) {
1055 return Error::deserialization("scalar must be varint");
1056 }
1057
1058 const auto res = Varint::decode(input, offset);
1059 if (!res) {
1060 return Error::deserialization("invalid varint");
1061 }
1062 offset += res->second;
1063
1064 T val;
1065 if constexpr (std::is_same_v<T, bool>) {
1066 val = (res->first != 0);
1067 } else if constexpr (std::is_floating_point_v<T>) {
1068 if constexpr (sizeof(T) == 4) {
1069 val = std::bit_cast<T>(static_cast<uint32_t>(res->first));
1070 } else {
1071 val = std::bit_cast<T>(res->first);
1072 }
1073 } else {
1074 val = std::bit_cast<T>(
1075 static_cast<std::make_unsigned_t<T>>(res->first));
1076 }
1077 // Deserialize will validate fields after the entire message is
1078 // deserialized
1079 field.set_without_validation(val);
1080 return std::nullopt;
1081 }

1194 template <typename FieldT>
1195 [[nodiscard]] static constexpr std::optional<Error>
1196 deserialize_map_elements(FieldT& field, std::span<const std::byte> input,
1197 std::size_t& offset,
1198 std::size_t end_offset) noexcept {
1199 using KeyFieldT = typename FieldT::PairType::first_type;
1200 using ValueFieldT = typename FieldT::PairType::second_type;
1201
1202 // Read count
1203 const auto count_res = Varint::decode(input, offset);
1204 if (!count_res) {
1205 return Error::deserialization("invalid map count");
1206 }
1207 offset += count_res->second;
1208 const std::size_t count = static_cast<std::size_t>(count_res->first);
1209
1210 // Deserialize key-value pairs
1211 for (std::size_t i = 0; i < count; ++i) {
1212 KeyFieldT key_field{};
1213 ValueFieldT val_field{};
1214
1215 if (const auto err = deserialize_value_without_tag<KeyFieldT>(
1216 key_field, input, offset)) {
1217 return err;
1218 }
1219 if (const auto err = deserialize_value_without_tag<ValueFieldT>(
1220 val_field, input, offset)) {
1221 return err;
1222 }
1223
1224 if (const auto err = field.insert(extract_map_value(key_field),
1225 extract_map_value(val_field))) {
1226 return err;
1227 }
1228 }
1229 return std::nullopt;

1283 template <typename FieldT>
1284 [[nodiscard]] static constexpr std::optional<Error> deserialize_field_value(
1285 FieldT& field, WireType wire_type, std::span<const std::byte> input,
1286 std::size_t& offset) noexcept {
1287 if constexpr (Crunch::messages::is_array_field_v<FieldT>) {
1288 return deserialize_array_field(field, wire_type, input, offset);
1289 } else if constexpr (Crunch::messages::is_map_field_v<FieldT>) {
1290 return deserialize_map_field(field, wire_type, input, offset);
1291 } else {
1292 using ValueType = typename detail::ext<FieldT>::type;
1293 if constexpr (Crunch::fields::is_scalar_v<ValueType>) {
1294 return deserialize_scalar_field(field, wire_type, input,
1295 offset);
1296 } else if constexpr (Crunch::fields::is_string_v<ValueType>) {
1297 return deserialize_string_field(field, wire_type, input,
1298 offset);
1299 } else if constexpr (Crunch::messages::HasCrunchMessageInterface<
1300 ValueType>) {
1301 return deserialize_nested_message_field(field, wire_type, input,
1302 offset);
1303 }
1304 }
1305 return std::nullopt;

1321 template <typename Tuple, std::size_t I = 0>
1322 [[nodiscard]] static constexpr std::optional<Error> visit_fields_tlv(
1323 Tuple&& fields_tuple, FieldId target_id, WireType wt,
1324 std::span<const std::byte> input, std::size_t& offset,
1325 bool& found) noexcept {
1326 using TupleT = std::remove_cvref_t<Tuple>;
1327 if constexpr (I < std::tuple_size_v<TupleT>) {
1328 auto& f = std::get<I>(fields_tuple);
1329 if (f.field_id == target_id) {
1330 found = true;
1331 return deserialize_field_value(f, wt, input, offset);
1332 }
1333 return visit_fields_tlv<Tuple, I + 1>(
1334 std::forward<Tuple>(fields_tuple), target_id, wt, input, offset,
1335 found);
1336 }
1337 return std::nullopt;
